 The addrinfo structure used by getaddrinfo() contains the following fields:

         struct addrinfo {
             int              ai_flags;
             int              ai_family;
             int              ai_socktype;
             int              ai_protocol;
             socklen_t        ai_addrlen;
             struct sockaddr *ai_addr;
             char            *ai_canonname;
             struct addrinfo *ai_next;
         };
